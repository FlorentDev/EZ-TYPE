99 /* Lis une chaine de caracteres de longueur donnee */
100 void lisChaine_deTaille(char *chaine, int taille)
101 {
102  char *ptrFinLigne;
103 
104  if (fgets(chaine, taille, stdin) != NULL)
105  {
106  if ((ptrFinLigne = strpbrk(chaine, "\n\r")) != NULL)
107  {
108  *ptrFinLigne = '\0'; /* Remplace le premier caractere de fin de
109  ligne par un caractere de fin de chaine */
110  }
111  else
112  {
113  chaine[taille-1] = '\0';
114  }
115  }
116  else
117  chaine[0] = '\0';
118 }

225 /* Renvoie une valeur aleatoire comprise entre 0 inclus et 1 exclu */
226 float valeurAleatoire(void)
227 {
228  if (graineGeneAleatoire == 0UL)
229  {
230  initialiseValeurAleatoire((float)(time(NULL)&0x07FFFFF));
231  }
232  return (float)rand()/((float)RAND_MAX+1);
233 }
