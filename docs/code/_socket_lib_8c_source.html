23 // Cree une nouvelle connexion qui pourra ensuite etre branchee sur une adresse et un port particulier
24 int nouvelleConnexion(void)
25 {
26  return socket(PF_INET, SOCK_STREAM, 0); // Ouverture d'un flux de donnees Internet

29 // Essaie de brancher une connexion sur une machine dont l'adresse et le port de communication sont donnes
30 bool ouvreConnexion_Avec_SurPort(int descripteurConnexion, const char *adresseIP, short port)
31 {
32  struct sockaddr_in adresseConnexion;
33 
34  adresseConnexion.sin_family = AF_INET; // La famille d'adressage ciblee est celle d'Internet
35  adresseConnexion.sin_port = htons(port); // Le port selectionne
36  adresseConnexion.sin_addr.s_addr = inet_addr(adresseIP); // L'adresse selectionne
37  memset(adresseConnexion.sin_zero, '\0', sizeof(adresseConnexion.sin_zero)); // Mise a zero obligatoire
38 
39  // On essaie de se connecter
40  return connect(descripteurConnexion, (struct sockaddr *)&adresseConnexion, sizeof(adresseConnexion)) == 0;
41 }

43 // Attend qu'une connexion tierce soit etablie sur la machine presente et sur le port donne en parametre
44 int attendConnexionSurPort(short port)
45 {
46  struct sockaddr_in adresseLocale;
47  struct sockaddr_in adresseClient;
48 #ifdef _WIN32
49  int sockaddr_in_size = sizeof(adresseClient);
50 #else
51  socklen_t sockaddr_in_size = (socklen_t)sizeof(adresseClient);
52 #endif
53  int descripteurConnexion;
54  int connexion = 0;
55 
56  descripteurConnexion = nouvelleConnexion();
57  if (descripteurConnexion > 0)
58  {
59  adresseLocale.sin_family = AF_INET;; // La famille d'adressage ciblee est celle d'Internet
60  adresseLocale.sin_port = htons(port); // Le port selectionne
61  adresseLocale.sin_addr.s_addr = htonl(INADDR_ANY); // Choix automatique de l'adresse IP locale
62  memset(adresseLocale.sin_zero, '\0', sizeof(adresseLocale.sin_zero)); // Mise a zero obligatoire
63 
64  if (bind(descripteurConnexion, (struct sockaddr *)&adresseLocale, sizeof(adresseLocale)) != -1) // On autorise le referencement de la connexion
65  {
66  if (listen(descripteurConnexion, 1) == 0) // Si on entend quelque chose lors de l'ecoute sur une seule et unique connexion
67  connexion = accept(descripteurConnexion, (struct sockaddr *)&adresseClient, &sockaddr_in_size); // On accepte le flux d'informations qui en arrive
68  }
69  fermeConnexion(descripteurConnexion);
70  }
71  return connexion;
72 }

88 // Envoie sur une connexion l'entier fourni (l'endianness est conservee)
89 // Renvoie faux si la totalite des octets formant l'entier n'a pas ete envoyee
90 bool envoieEntier_Sur(int entier, int descripteurConnexion)
91 {
92  entier = htonl(entier);
93  return sizeof(entier) == envoie_DeTaille_Sur(&entier, sizeof(entier), descripteurConnexion); // Erreur si la taille envoyee n'est pas egale a celle de la donnee
94 }

96 // Receptionne a partir d'une connexion un entier dans une zone de memoire passee en parametre
97 // Renvoie faux si la totalite des octets formant l'entier n'a pas ete receptionnee
98 bool receptionneEntier_Sur(int *entier, int descripteurConnexion)
99 {
100  int tailleLue = receptionne_DeTaille_Sur(entier, sizeof(int), descripteurConnexion);
101  *entier = ntohl(*entier);
102  return sizeof(int) == tailleLue; // Erreur si la taille lue n'est pas celle attendue
103 }
