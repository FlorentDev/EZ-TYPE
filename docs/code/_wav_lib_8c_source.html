55

72  // S'il n'y a pas de compression (encodage PCM)
73  if (little16VersNatif(buffer2o) == PCM_ENCODING)

83  fread(&ID, 1, sizeof(ID), fichierWAV);
84  // Si ce qui suit contient bien 'data'

105  {

158  ID[2] = 't';
159  ID[3] = ' ';
