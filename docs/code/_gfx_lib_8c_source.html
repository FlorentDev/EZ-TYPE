212 {
213  glutInitWindowPosition(xCoinHautGauche, yCoinHautGauche);
214  glutInitWindowSize(largeur, hauteur);
215  sIDWindow = glutCreateWindow(nom);

217  /* Les fonctions callback doivent etre fixees apres creation de la fenetre,
218  c'est pourquoi on memorise juste les valeurs par defaut ci-apres; cela permet
219  de plus de personnaliser les callback apres coup: les callback peuvent etre
220  deroutees de leurs valeurs par defaut si besoin est, il suffit d'appeler les
221  fonctions adequates avant de lancer la boucle des evenements */
222  if (sFonctionAffichage == NULL) sFonctionAffichage = fonctionAffichage;
223  /* La fonction de gestion d'inactivite n'est pas geree par defaut car elle consomme trop de ressources */

232  glHint(GL_PERSPECTIVE_CORRECTION_HINT, GL_NICEST); /* Correction de perspective */

447 /* Lis une portion de la fenetre, couvrant largeur*hauteur et demarrant en (x, y)
448 Les donnees lues sont sauvees comme une succession de valeurs B, V, R de type
449 unsigned char */
450 void lisImage(int x, int y, int largeur, int hauteur, unsigned char *donnees)
451 {
452  unsigned char *ptrDonnees;
453  int *pixels = (int*)malloc(largeur*hauteur*sizeof(int));
454  int *ptrPixel;
455  int index;
456 
457  glReadPixels(x, y, largeur, hauteur, GL_BGRA, GL_UNSIGNED_INT_8_8_8_8_REV, pixels);
458 
459  ptrPixel = pixels;
460  ptrDonnees = donnees;
461  for
462  (index = largeur*hauteur; index != 0; --index) /* On parcourt tous les pixels lus */
463  {
464  int pixel = *ptrPixel; /* On lit le pixel courant */
465 
466  /* On extrait chaque composante du pixel, qu'on met dans les donnees */
467  ptrDonnees[0] = bleuDuPixel(pixel);
468  ptrDonnees[1] = vertDuPixel(pixel);
469  ptrDonnees[2] = rougeDuPixel(pixel);
470 
471  ++ptrPixel; /* On passe au pixel suivant */
472  ptrDonnees += 3; /* On passe aux donnees suivantes */
473  }
474 
475  free(pixels);
476 }

478 // Convertit largeur*hauteur pixels de la forme bleu, vert, rouge en entiers opacite, rouge, vert, bleu little endian
479 static int *BVR2ARVB(int largeur, int hauteur, const unsigned char *donnees)
480 {
481  const unsigned char *ptrDonnees;
482  unsigned char *pixels = (unsigned char*)malloc(largeur*hauteur*sizeof(int));
483  unsigned char *ptrPixel;
484  int index;
485 
486  ptrPixel = pixels;
487  ptrDonnees = donnees;
488  for
489  (index = largeur*hauteur; index != 0; --index) /* On parcourt tous les
490  pixels de l'image */
491  {
492  ptrPixel[0] = ptrDonnees[0];
493  ptrPixel[1] = ptrDonnees[1];
494  ptrPixel[2] = ptrDonnees[2];
495  // If the pixel is black, set the alpha to 0
496  if(ptrPixel[0] == 0 && ptrPixel[1] == 0 && ptrPixel[2] == 0) {
497  ptrPixel[3] = 0x000;
498  } else {
499  ptrPixel[3] = 0x0FF;
500  }
501  ptrDonnees += 3; /* On passe a la premiere composante du pixel suivant */
502  ptrPixel += 4; /* On passe au pixel suivant */
503  }
504  return (int*)pixels;
505 }

507 // Ecris dans la fenetre une image BVR couvrant largeur*hauteur et demarrant en (x, y) */

527 // Cree une texture reutilisable de largeur et hauteur donnees, faites de pixels ABVR little endian
528 Texture2D *creeTexture2D(int largeur, int hauteur, const int *pixels)
529 {
530  Texture2D *texture = (Texture2D*)malloc(sizeof(Texture2D));
531  int largeur2m = 64;
532  int hauteur2m = 64;
533 
534  texture->largeur = largeur;
535  texture->hauteur = hauteur;
536  while(largeur2m < largeur) largeur2m += largeur2m; // On calcule la largeur sous la forme 2^m
537  while(hauteur2m < hauteur) hauteur2m += hauteur2m; // On calcule la hauteur sous la forme 2^m
538  texture->proportionX = (float)largeur/(float)largeur2m;
539  texture->proportionY = (float)hauteur/(float)hauteur2m;
540 
541  texture->donnees = (int*)calloc(largeur2m*hauteur2m, sizeof(int));
542 
543  glPixelStorei(GL_UNPACK_ALIGNMENT, 1);
544  glGenTextures(1, &texture->numero);
545  glBindTexture(GL_TEXTURE_2D, texture->numero);
546  glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_WRAP_S, GL_REPEAT);
547  glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_WRAP_T, GL_REPEAT);
548  glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MAG_FILTER, GL_NEAREST);
549  glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MIN_FILTER, GL_NEAREST);
550  glTexImage2D(GL_TEXTURE_2D, 0, GL_RGBA, largeur2m, hauteur2m, 0, GL_RGBA, GL_UNSIGNED_BYTE, texture->donnees);
551  modifieTexture2D(texture, pixels);
552 
553  return texture;
554 }
