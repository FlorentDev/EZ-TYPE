37 /* Renvoine la longueur d'une scanline, qui doit etre multiple de quatre octets */
38 static int tailleScanLineRGB(int largeurImage)

43 /* Fonction essayant de lire le fichier passe en parametre, et renvoyant une structure
44  contenant les informations de l'image en cas de succes, NULL sinon */
45 DonneesImageRGB *lisBMPRGB(char *nom)
46 {
47  DonneesImageRGB *donneesImage = (DonneesImageRGB*)calloc(1, sizeof(DonneesImageRGB));
48  bool toutOK = false;
49 
50  if
51  (donneesImage != NULL)
52  {
53  /* donneesImage != NULL */
54 
55  FILE *fichierBMP;
56 
57  if
58  ((fichierBMP = fopen(nom, "rb")) != NULL)
59  {
60  /* fichierBMP != NULL */
61 
62  /* Si le header commence bien par 'B' et 'M'... */
63  if
64  (fgetc(fichierBMP) == 'B' && fgetc(fichierBMP) == 'M')
65  {
66  /* ... alors la suite du fichier peut etre interpretee */
67  int fileHeader[3];
68 
69  if
70  (fread(fileHeader, sizeof(fileHeader), 1, fichierBMP) == 1)
71  {
72  /* Lecture du header de fichier */
73  int offsetDonnees = little32VersNatif(fileHeader[2]);
74 
75  int bitmapInfoHeader[10];
76  if
77  (fread(bitmapInfoHeader, sizeof(bitmapInfoHeader), 1, fichierBMP) == 1)
78  {
79  /* On ne lit que les images 24 bits non compressees */
80  if
81  ((little32VersNatif(bitmapInfoHeader[3]) == 0x00180001) && (little32VersNatif(bitmapInfoHeader[4]) == 0))
82  {
83  bool hautVersBas = false; // Utile pour detecter les BMP allant de haut en bas au lieu de bas vers haut
84  /* Lecture du header de bitmap (uniquement les informations qui nous interessent) */
85  donneesImage->largeurImage = little32VersNatif(bitmapInfoHeader[1]);
86  donneesImage->hauteurImage = little32VersNatif(bitmapInfoHeader[2]);
87 
88  if (donneesImage->hauteurImage < 0)
89  {
90  donneesImage->hauteurImage = -donneesImage->hauteurImage;
91  hautVersBas = true;
92  }
93 
94  /* On alloue la place pour lire les donnes */
95  if
96  ((donneesImage->donneesRGB = (unsigned char *)malloc((unsigned int)donneesImage->largeurImage*(unsigned int)donneesImage->hauteurImage*3)) != NULL)
97  {
98  /* donneesImage->donneesRGB != NULL */
99 
100  unsigned char *scanline;
101  /* Une scanline doit avoir une taille multiple de quatre octets */
102  int tailleScanLine = tailleScanLineRGB(donneesImage->largeurImage);
103 
104  /* On alloue la place pour lire chaque scanline */
105  if
106  ((scanline = (unsigned char *)malloc((unsigned int)tailleScanLine)) != NULL)
107  {
108  /* scanline != NULL */
109 
110  /* On se positionne sur le debut des donnees a lire */
111  if
112  (fseek(fichierBMP, offsetDonnees, SEEK_SET) == 0)
113  {
114  int indexLigne;
115  unsigned char *pointeurDonnees;
116  pointeurDonnees = donneesImage->donneesRGB +
117  (hautVersBas ?
118  donneesImage->largeurImage*3*(donneesImage->hauteurImage-1) :
119  0);
120 
121  /* Tout s'est bien passe jusqu'a present */
122  toutOK = true;
123  for
124  (indexLigne = 0; (indexLigne < donneesImage->hauteurImage) && toutOK; ++indexLigne)
125  {
126  /* On recopie ligne a ligne les informations */
127  if
128  (fread(scanline, (unsigned int)tailleScanLine, 1, fichierBMP) == 1)
129  {
130  memcpy(pointeurDonnees, scanline, (unsigned int)(donneesImage->largeurImage*3));
131  if (hautVersBas)
132  pointeurDonnees -= donneesImage->largeurImage*3;
133  else
134  pointeurDonnees += donneesImage->largeurImage*3;
135  }
136  else
137  {
138  toutOK = false;
139  }
140  }
141  }
142  /* Quoi qu'il arrive il faut liberer le buffer memoire de lecture d'une scanline */
143  free(scanline);
144  }
145 
146  /* Si tout ne s'est pas bien passe, il faut liberer l'espace memoire pour stocker l'image */
147  if
148  (!toutOK)
149  {
150  free(donneesImage->donneesRGB);
151  }
152  }
153  }
154  }
155  }
156  }
157 
158 
159  /* Quoi qu'il arrive il faut fermer le fichier */
160  fclose(fichierBMP);
161  }
162 
163  /* Si tout ne s'est pas bien passe on libere les donnees image et on les met a NULL */
164  if
165  (!toutOK)
166  {
167  free(donneesImage);
168  donneesImage = NULL;
169  }
170  }
171 
172  return donneesImage;
173 }

175 /* Fonction ecrivant les informations de l'image dans le fichier passe en parametre.
176  Renvoie faux en cas de probleme, vrai sinon */
177 bool ecrisBMPRGB_Dans(DonneesImageRGB *donneesImage, char *nom)
178 {
179  FILE *fichierBMP;
180  bool toutOK = false;
181 
182  if
183  ((fichierBMP = fopen(nom, "wb")) != NULL)
184  {
185  /* fichierBMP != NULL */
186 
187  /* Le header commence par 'B' et 'M' */
188  if
189  (fwrite("BM", 2, 1, fichierBMP) == 1)
190  {
191  int headerBMP[14]; /* Sert a memoriser toutes les informations relatives au BMP */
192  int tailleScanLine = tailleScanLineRGB(donneesImage->largeurImage);
193 
194  memset(headerBMP, 0, 14*4);
195  /* Taille du fichier */
196  headerBMP[0] = natif32VersLittle(2+12+40+4+tailleScanLine*donneesImage->hauteurImage);
197  headerBMP[1] = 0;
198  /* Offset a partir duquel commencent les donnees */
199  headerBMP[2] = natif32VersLittle(2+12+40+4);
200  /* Taille de la structure BITMAPINFOHEADER qui doit faire 40 octets */
201  headerBMP[3] = natif32VersLittle(40);
202  /* Largeur de l'image */
203  headerBMP[4] = natif32VersLittle(donneesImage->largeurImage);
204  /* Hauteur de l'image */
205  headerBMP[5] = natif32VersLittle(donneesImage->hauteurImage);
206  /* Image monoplan de 24 bits */
207  headerBMP[6] = natif32VersLittle(0x00180001);
208  /* Image non compressee */
209  headerBMP[7] = 0;
210  /* Les six champs suivants ne nous sont pas utiles */
211  headerBMP[8] = 0;
212  headerBMP[9] = 0;
213  headerBMP[10] = 0;
214  headerBMP[11] = 0;
215  headerBMP[12] = 0;
216  headerBMP[13] = 0;
217 
218  /* Ecriture du header de fichier */
219  if
220  (fwrite(headerBMP, 14*4, 1, fichierBMP) == 1)
221  {
222  unsigned char *scanline;
223 
224  /* On alloue la place pour ecrire chaque scanline */
225  if
226  ((scanline = (unsigned char *)malloc((unsigned int)tailleScanLine)) != NULL)
227  {
228  /* scanline != NULL */
229 
230  int indexLigne;
231  unsigned char *pointeurDonnees = donneesImage->donneesRGB;
232 
233  /* Tout s'est bien passe jusqu'a present */
234  toutOK = true;
235 
236  for
237  (indexLigne = 0; (indexLigne < donneesImage->hauteurImage) && toutOK; ++indexLigne)
238  {
239  /* On recopie ligne a ligne les informations */
240  memcpy(scanline, pointeurDonnees, (unsigned int)(donneesImage->largeurImage*3));
241  pointeurDonnees += donneesImage->largeurImage*3;
242 
243  if
244  (fwrite(scanline, (unsigned int)tailleScanLine, 1, fichierBMP) != 1)
245  {
246  toutOK = false;
247  }
248  }
249 
250  /* Quoi qu'il arrive il faut liberer le buffer memoire d'ecriture d'une scanline */
251  free(scanline);
252  }
253  }
254  }
255 
256 
257  /* Quoi qu'il arrive il faut fermer le fichier */
258  fclose(fichierBMP);
259  }
260 
261  return toutOK;
262 }
